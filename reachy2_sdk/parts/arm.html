181    def turn_off_smoothly(self) -> None:
182        """Turn all motors of the part off.
183
184        All arm's motors will see their torque limit reduces for 3 seconds, then will be fully compliant.
185        """
186        torque_limit_low = 35
187        torque_limit_high = 100
188        duration = 3
189
190        self.set_torque_limits(torque_limit_low)
191        self.goto_posture(duration=duration, wait_for_goto_end=False)
192
193        countingTime = 0
194        while countingTime < duration:
195            time.sleep(1)
196            torque_limit_low -= 10
197            self.set_torque_limits(torque_limit_low)
198            countingTime += 1
199
200        super().turn_off()
201        self.set_torque_limits(torque_limit_high)

222    def forward_kinematics(
223        self, joints_positions: Optional[List[float]] = None, degrees: bool = True
224    ) -> npt.NDArray[np.float64]:
225        """Compute the forward kinematics of the arm.
226
227        It will return the pose 4x4 matrix (as a numpy array) expressed in Reachy coordinate systems.
228        You can either specify a given joints position, otherwise it will use the current robot position.
229        """
230        req_params = {
231            "id": self._part_id,
232        }
233        if joints_positions is None:
234            present_joints_positions = [
235                joint.present_position for orbita in self._actuators.values() for joint in orbita._joints.values()
236            ]
237            req_params["position"] = list_to_arm_position(present_joints_positions, degrees)
238
239        else:
240            if len(joints_positions) != 7:
241                raise ValueError(f"joints_positions should be length 7 (got {len(joints_positions)} instead)!")
242            req_params["position"] = list_to_arm_position(joints_positions, degrees)
243        req = ArmFKRequest(**req_params)
244        resp = self._stub.ComputeArmFK(req)
245        if not resp.success:
246            raise ValueError(f"No solution found for the given joints ({joints_positions})!")
247
248        return np.array(resp.end_effector.pose.data).reshape((4, 4))

250    def inverse_kinematics(
251        self,
252        target: npt.NDArray[np.float64],
253        q0: Optional[List[float]] = None,
254        degrees: bool = True,
255        round: Optional[int] = None,
256    ) -> List[float]:
257        """Compute the inverse kinematics of the arm.
258
259        Given a pose 4x4 target matrix (as a numpy array) expressed in Reachy coordinate systems,
260        it will try to compute a joint solution to reach this target (or get close).
261
262        It will raise a ValueError if no solution is found.
263
264        You can also specify a basic joint configuration as a prior for the solution.
265        """
266        if target.shape != (4, 4):
267            raise ValueError("target shape should be (4, 4) (got {target.shape} instead)!")
268
269        if q0 is not None and (len(q0) != 7):
270            raise ValueError(f"q0 should be length 7 (got {len(q0)} instead)!")
271
272        if isinstance(q0, np.ndarray) and len(q0.shape) > 1:
273            raise ValueError("Vectorized kinematics not supported!")
274
275        req_params = {
276            "target": ArmEndEffector(
277                pose=Matrix4x4(data=target.flatten().tolist()),
278            ),
279            "id": self._part_id,
280        }
281
282        if q0 is not None:
283            req_params["q0"] = list_to_arm_position(q0, degrees)
284
285        else:
286            present_joints_positions = [
287                joint.present_position for orbita in self._actuators.values() for joint in orbita._joints.values()
288            ]
289            req_params["q0"] = list_to_arm_position(present_joints_positions, degrees)
290
291        req = ArmIKRequest(**req_params)
292        resp = self._stub.ComputeArmIK(req)
293
294        if not resp.success:
295            raise ValueError(f"No solution found for the given target ({target})!")
296
297        answer: List[float] = arm_position_to_list(resp.arm_position, degrees)
298        if round is not None:
299            answer = np.round(answer, round).tolist()
300        return answer

320    def goto_from_matrix(
321        self,
322        target: npt.NDArray[np.float64],
323        duration: float = 2,
324        wait: bool = False,
325        interpolation_mode: str = "minimum_jerk",
326        q0: Optional[List[float]] = None,
327    ) -> GoToId:
328        """Move the arm to a matrix target (or get close).
329
330        Given a pose 4x4 target matrix (as a numpy array) expressed in Reachy coordinate systems,
331        it will try to compute a joint solution to reach this target (or get close),
332        and move to this position in the defined duration.
333        """
334        if target.shape != (4, 4):
335            raise ValueError("target shape should be (4, 4) (got {target.shape} instead)!")
336        if q0 is not None and (len(q0) != 7):
337            raise ValueError(f"q0 should be length 7 (got {len(q0)} instead)!")
338        if duration == 0:
339            raise ValueError("duration cannot be set to 0.")
340        if self.is_off():
341            self._logger.warning(f"{self._part_id.name} is off. Goto not sent.")
342            return GoToId(id=-1)
343
344        if q0 is not None:
345            q0 = list_to_arm_position(q0)
346            request = GoToRequest(
347                cartesian_goal=CartesianGoal(
348                    arm_cartesian_goal=ArmCartesianGoal(
349                        id=self._part_id,
350                        goal_pose=Matrix4x4(data=target.flatten().tolist()),
351                        duration=FloatValue(value=duration),
352                        q0=q0,
353                    )
354                ),
355                interpolation_mode=get_grpc_interpolation_mode(interpolation_mode),
356            )
357        else:
358            request = GoToRequest(
359                cartesian_goal=CartesianGoal(
360                    arm_cartesian_goal=ArmCartesianGoal(
361                        id=self._part_id,
362                        goal_pose=Matrix4x4(data=target.flatten().tolist()),
363                        duration=FloatValue(value=duration),
364                    )
365                ),
366                interpolation_mode=get_grpc_interpolation_mode(interpolation_mode),
367            )
368        response = self._goto_stub.GoToCartesian(request)
369        if response.id == -1:
370            self._logger.error(f"Target pose:\n {target} \nwas not reachable. No command sent.")
371        elif wait:
372            self._wait_goto(response)
373        return response

375    def send_cartesian_interpolation(
376        self,
377        target: npt.NDArray[np.float64],
378        duration: float = 2,
379        arc_direction: Optional[str] = None,
380        elliptic_radius: Optional[float] = None,
381        interpolation_frequency: float = 120,
382        precision_distance_xyz: float = 0.003,
383    ) -> None:
384        """Move the arm to a matrix target (or get close).
385
386        Given a pose 4x4 target matrix (as a numpy array) expressed in Reachy coordinate systems,
387        it will try to compute a joint solution to reach this target (or get close).
388        It will interpolate the movement in cartesian space in a number of intermediate points defined
389        by the interpolation frequency and the duration.
390
391        Arguments :
392            target          : the 4x4 goal pose matrix in the robot coordinate system, reached at the end of the interpolation
393            duration        : the target duration of the movement
394            arc_direction   : if None, leads to a linear interpolation. If set, direction of the elliptic interpolation.
395                            Can be set to 'above', 'below', 'right', 'left', 'front' or 'back'
396            elliptic_radius : the second radius of the computed ellipse, first radius being the present to target pose distance.
397                            If None, leads to a circular interpolation.
398            interpolation_frequency : the frequency of the interpolation
399            precision_distance_xyz  : the maximum distance between the target pose and the reached pose after the interpolation.
400                                      Precision is prioritized over duration.
401        """
402
403        self.cancel_all_goto()
404        if target.shape != (4, 4):
405            raise ValueError("target shape should be (4, 4) (got {target.shape} instead)!")
406        if duration == 0:
407            raise ValueError("duration cannot be set to 0.")
408        if self.is_off():
409            self._logger.warning(f"{self._part_id.name} is off. Commands not sent.")
410            return
411        try:
412            self.inverse_kinematics(target)
413        except ValueError:
414            raise ValueError(f"Target pose: \n{target}\n is not reachable!")
415
416        origin_matrix = self.forward_kinematics()
417        nb_steps = int(duration * interpolation_frequency)
418        time_step = duration / nb_steps
419
420        q1, trans1 = decompose_matrix(origin_matrix)
421        q2, trans2 = decompose_matrix(target)
422
423        if arc_direction is None:
424            self._send_linear_interpolation(trans1, trans2, q1, q2, nb_steps=nb_steps, time_step=time_step)
425
426        else:
427            self._send_elliptical_interpolation(
428                trans1,
429                trans2,
430                q1,
431                q2,
432                arc_direction=arc_direction,
433                secondary_radius=elliptic_radius,
434                nb_steps=nb_steps,
435                time_step=time_step,
436            )
437
438        current_pose = self.forward_kinematics()
439        current_precision_distance_xyz = np.linalg.norm(current_pose[:3, 3] - target[:3, 3])
440        if current_precision_distance_xyz > precision_distance_xyz:
441            request = ArmCartesianGoal(
442                id=self._part_id,
443                goal_pose=Matrix4x4(data=target.flatten().tolist()),
444            )
445            self._stub.SendArmCartesianGoal(request)
446            time.sleep(time_step)
447
448            current_pose = self.forward_kinematics()
449            current_precision_distance_xyz = np.linalg.norm(current_pose[:3, 3] - target[:3, 3])
450        self._logger.info(f"l2 xyz distance to goal: {current_precision_distance_xyz}")
451
452    def _send_linear_interpolation(
453        self,
454        origin_trans: npt.NDArray[np.float64],
455        target_trans: npt.NDArray[np.float64],
456        origin_rot: Quaternion,
457        target_rot: Quaternion,
458        nb_steps: int,
459        time_step: float,
460    ) -> None:
461        """Generate linear interpolation."""
462        for t in np.linspace(0, 1, nb_steps):
463            # Linear interpolation for translation
464            trans_interpolated = (1 - t) * origin_trans + t * target_trans
465
466            # SLERP for rotation interpolation
467            q_interpolated = Quaternion.slerp(origin_rot, target_rot, t)
468            rot_interpolated = q_interpolated.rotation_matrix
469
470            # Recompose the interpolated matrix
471            interpolated_matrix = recompose_matrix(rot_interpolated, trans_interpolated)
472
473            request = ArmCartesianGoal(
474                id=self._part_id,
475                goal_pose=Matrix4x4(data=interpolated_matrix.flatten().tolist()),
476            )
477            self._stub.SendArmCartesianGoal(request)
478            time.sleep(time_step)

480    def _send_elliptical_interpolation(
481        self,
482        origin_trans: npt.NDArray[np.float64],
483        target_trans: npt.NDArray[np.float64],
484        origin_rot: Quaternion,
485        target_rot: Quaternion,
486        arc_direction: str,
487        secondary_radius: Optional[float],
488        nb_steps: int,
489        time_step: float,
490    ) -> None:
491        """Generate elliptical interpolation."""
492        vector_target_origin = target_trans - origin_trans
493
494        center = (origin_trans + target_trans) / 2
495        radius = float(np.linalg.norm(vector_target_origin) / 2)
496
497        vector_origin_center = origin_trans - center
498        vector_target_center = target_trans - center
499
500        if np.isclose(radius, 0, atol=1e-03):
501            self._logger.warning(f"{self._part_id.name} is already at the target pose. No command sent.")
502            return
503        if secondary_radius is None:
504            secondary_radius = radius
505        if secondary_radius is not None and secondary_radius > 0.3:
506            self._logger.warning("interpolation elliptic_radius was too large, reduced to 0.3")
507            secondary_radius = 0.3
508
509        normal = get_normal_vector(vector=vector_target_origin, arc_direction=arc_direction)
510
511        if normal is None:
512            self._logger.warning("arc_direction has no solution. Executing linear interpolation instead.")
513            self._send_linear_interpolation(
514                origin_trans=origin_trans,
515                target_trans=target_trans,
516                origin_rot=origin_rot,
517                target_rot=target_rot,
518                nb_steps=nb_steps,
519                time_step=time_step,
520            )
521            return
522
523        cos_angle = np.dot(vector_origin_center, vector_target_center) / (
524            np.linalg.norm(vector_origin_center) * np.linalg.norm(vector_target_center)
525        )
526        angle = np.arccos(np.clip(cos_angle, -1, 1))
527
528        for t in np.linspace(0, 1, nb_steps):
529            # Interpolated angles
530            theta = t * angle
531
532            # Rotation of origin_vector around the circle center in the plan defined by 'normal'
533            q1 = Quaternion(axis=normal, angle=theta)
534            rotation_matrix = q1.rotation_matrix
535
536            # Interpolated point in plan
537            trans_interpolated = np.dot(rotation_matrix, vector_origin_center)
538            # Adjusting the ellipse
539            ellipse_interpolated = trans_interpolated * np.array([1, 1, secondary_radius / radius])
540            trans_interpolated = ellipse_interpolated + center
541
542            # SLERP for the rotation
543            q_interpolated = Quaternion.slerp(origin_rot, target_rot, t)
544            rot_interpolated = q_interpolated.rotation_matrix
545
546            # Recompose the interpolated matrix
547            interpolated_matrix = recompose_matrix(rot_interpolated, trans_interpolated)
548
549            request = ArmCartesianGoal(
550                id=self._part_id,
551                goal_pose=Matrix4x4(data=interpolated_matrix.flatten().tolist()),
552            )
553            self._stub.SendArmCartesianGoal(request)
554            time.sleep(time_step)

556    def goto_joints(
557        self,
558        positions: List[float],
559        duration: float = 2,
560        wait: bool = False,
561        interpolation_mode: str = "minimum_jerk",
562        degrees: bool = True,
563    ) -> GoToId:
564        """Move the arm's joints to reach the given position.
565
566        Given a list of joint positions (exactly 7 joint positions),
567        it will move the arm to that position.
568        """
569        if len(positions) != 7:
570            raise ValueError(f"positions should be of length 7 (got {len(positions)} instead)!")
571        if duration == 0:
572            raise ValueError("duration cannot be set to 0.")
573        if self.is_off():
574            self._logger.warning(f"{self._part_id.name} is off. Goto not sent.")

578        request = GoToRequest(

582            interpolation_mode=get_grpc_interpolation_mode(interpolation_mode),
583        )
584        response = self._goto_stub.GoToJoints(request)
585        if response.id == -1:
586            self._logger.error(f"Position {positions} was not reachable. No command sent.")
587        elif wait:
588            self._wait_goto(response)

376    def send_cartesian_interpolation(
377        self,
378        target: npt.NDArray[np.float64],
379        duration: float = 2,
380        arc_direction: Optional[str] = None,
381        elliptic_radius: Optional[float] = None,
382        interpolation_frequency: float = 120,
383        precision_distance_xyz: float = 0.003,
384    ) -> None:
385        """Move the arm to a matrix target (or get close).
386
387        Given a pose 4x4 target matrix (as a numpy array) expressed in Reachy coordinate systems,
388        it will try to compute a joint solution to reach this target (or get close).
389        It will interpolate the movement in cartesian space in a number of intermediate points defined
390        by the interpolation frequency and the duration.
391
392        Arguments :
393            target          : the 4x4 goal pose matrix in the robot coordinate system, reached at the end of the interpolation
394            duration        : the target duration of the movement
395            arc_direction   : if None, leads to a linear interpolation. If set, direction of the elliptic interpolation.
396                            Can be set to 'above', 'below', 'right', 'left', 'front' or 'back'
397            elliptic_radius : the second radius of the computed ellipse, first radius being the present to target pose distance.
398                            If None, leads to a circular interpolation.
399            interpolation_frequency : the frequency of the interpolation
400            precision_distance_xyz  : the maximum distance between the target pose and the reached pose after the interpolation.
401                                      Precision is prioritized over duration.
402        """
403
404        self.cancel_all_goto()
405        if target.shape != (4, 4):
406            raise ValueError("target shape should be (4, 4) (got {target.shape} instead)!")
407        if duration == 0:
408            raise ValueError("duration cannot be set to 0.")
409        if self.is_off():
410            self._logger.warning(f"{self._part_id.name} is off. Commands not sent.")
411            return
412        try:
413            self.inverse_kinematics(target)
414        except ValueError:
415            raise ValueError(f"Target pose: \n{target}\n is not reachable!")
416
417        origin_matrix = self.forward_kinematics()
418        nb_steps = int(duration * interpolation_frequency)
419        time_step = duration / nb_steps
420
421        q1, trans1 = decompose_matrix(origin_matrix)
422        q2, trans2 = decompose_matrix(target)
423
424        if arc_direction is None:
425            self._send_linear_interpolation(trans1, trans2, q1, q2, nb_steps=nb_steps, time_step=time_step)
426
427        else:
428            self._send_elliptical_interpolation(
429                trans1,
430                trans2,
431                q1,
432                q2,
433                arc_direction=arc_direction,
434                secondary_radius=elliptic_radius,
435                nb_steps=nb_steps,
436                time_step=time_step,
437            )
438
439        current_pose = self.forward_kinematics()
440        current_precision_distance_xyz = np.linalg.norm(current_pose[:3, 3] - target[:3, 3])
441        if current_precision_distance_xyz > precision_distance_xyz:
442            request = ArmCartesianGoal(
443                id=self._part_id,
444                goal_pose=Matrix4x4(data=target.flatten().tolist()),
445            )
446            self._stub.SendArmCartesianGoal(request)
447            time.sleep(time_step)

460        time_step: float,
461    ) -> None:

481    def _send_elliptical_interpolation(
482        self,
483        origin_trans: npt.NDArray[np.float64],
484        target_trans: npt.NDArray[np.float64],
485        origin_rot: Quaternion,
486        target_rot: Quaternion,
487        arc_direction: str,
488        secondary_radius: Optional[float],
489        nb_steps: int,
490        time_step: float,
491    ) -> None:
492        """Generate elliptical interpolation."""
493        vector_target_origin = target_trans - origin_trans
494
495        center = (origin_trans + target_trans) / 2
496        radius = float(np.linalg.norm(vector_target_origin) / 2)
497
498        vector_origin_center = origin_trans - center
499        vector_target_center = target_trans - center
500
501        if np.isclose(radius, 0, atol=1e-03):
502            self._logger.warning(f"{self._part_id.name} is already at the target pose. No command sent.")
503            return
504        if secondary_radius is None:
505            secondary_radius = radius
506        if secondary_radius is not None and secondary_radius > 0.3:
507            self._logger.warning("interpolation elliptic_radius was too large, reduced to 0.3")
508            secondary_radius = 0.3
509
510        normal = get_normal_vector(vector=vector_target_origin, arc_direction=arc_direction)
511
512        if normal is None:
513            self._logger.warning("arc_direction has no solution. Executing linear interpolation instead.")
514            self._send_linear_interpolation(
515                origin_trans=origin_trans,
516                target_trans=target_trans,
517                origin_rot=origin_rot,
518                target_rot=target_rot,
519                nb_steps=nb_steps,
520                time_step=time_step,
521            )
522            return
523
524        cos_angle = np.dot(vector_origin_center, vector_target_center) / (
525            np.linalg.norm(vector_origin_center) * np.linalg.norm(vector_target_center)
526        )
527        angle = np.arccos(np.clip(cos_angle, -1, 1))
528
529        for t in np.linspace(0, 1, nb_steps):
530            # Interpolated angles
531            theta = t * angle
532
533            # Rotation of origin_vector around the circle center in the plan defined by 'normal'
534            q1 = Quaternion(axis=normal, angle=theta)
535            rotation_matrix = q1.rotation_matrix
536
537            # Interpolated point in plan
538            trans_interpolated = np.dot(rotation_matrix, vector_origin_center)
539            # Adjusting the ellipse
540            ellipse_interpolated = trans_interpolated * np.array([1, 1, secondary_radius / radius])
541            trans_interpolated = ellipse_interpolated + center
542
543            # SLERP for the rotation
544            q_interpolated = Quaternion.slerp(origin_rot, target_rot, t)
545            rot_interpolated = q_interpolated.rotation_matrix
546
547            # Recompose the interpolated matrix
548            interpolated_matrix = recompose_matrix(rot_interpolated, trans_interpolated)
549
550            request = ArmCartesianGoal(
551                id=self._part_id,
552                goal_pose=Matrix4x4(data=interpolated_matrix.flatten().tolist()),
553            )
554            self._stub.SendArmCartesianGoal(request)
555            time.sleep(time_step)
