 39class Head(JointsBasedPart, IGoToBasedPart):
 40    """Head class for controlling the head of Reachy.
 41
 42    The `Head` class manages the neck actuator and provides utilities for controlling the orientation
 43    of the head, such as moving to a specific posture or looking at a Cartesian point in Reachy's
 44    coordinate system.
 45
 46    Attributes:
 47        neck: An instance of `Orbita3d` representing the neck actuator of the head.
 48    """

 50    def __init__(
 51        self,
 52        head_msg: Head_proto,
 53        initial_state: HeadState,
 54        grpc_channel: grpc.Channel,
 55        goto_stub: GoToServiceStub,
 56    ) -> None:
 57        """Initialize the Head component with its actuators.
 58
 59        Sets up the necessary attributes and configuration for the head, including the gRPC
 60        stubs and initial state.
 61
 62        Args:
 63            head_msg: The Head_proto object containing the configuration details for the head.
 64            initial_state: The initial state of the head, represented as a HeadState object.
 65            grpc_channel: The gRPC channel used to communicate with the head's gRPC service.
 66            goto_stub: The GoToServiceStub used to handle goto-based movements for the head.
 67        """
 68        JointsBasedPart.__init__(self, head_msg, grpc_channel, HeadServiceStub(grpc_channel))
 69        IGoToBasedPart.__init__(self, self, goto_stub)
 70
 71        self._setup_head(head_msg, initial_state)
 72        self._actuators = {
 73            "neck": self.neck,
 74        }

 76    def _setup_head(self, head: Head_proto, initial_state: HeadState) -> None:
 77        """Set up the head with its actuators.
 78
 79        This method initializes the neck and antenna actuators for the head and sets their initial state.
 80
 81        Args:
 82            head: A Head_proto object containing the configuration details for the head.
 83            initial_state: A HeadState object representing the initial state of the head's actuators.
 84        """
 85        description = head.description
 86        self._neck = Orbita3d(
 87            uid=description.neck.id.id,
 88            name=description.neck.id.name,
 89            initial_state=initial_state.neck_state,
 90            grpc_channel=self._grpc_channel,
 91            part=self,
 92            joints_position_order=[NeckJoints.ROLL, NeckJoints.PITCH, NeckJoints.YAW],
 93        )

 95    def __repr__(self) -> str:
 96        """Clean representation of an Head."""
 97        s = "\n\t".join([act_name + ": " + str(actuator) for act_name, actuator in self._actuators.items()])
 98        return f"""<Head on={self.is_on()} actuators=\n\t{
 99            s
100        }\n>"""

127    def look_at(
128        self, x: float, y: float, z: float, duration: float = 2.0, wait: bool = False, interpolation_mode: str = "minimum_jerk"
129    ) -> GoToId:
130        """Compute and send a neck position to look at a specified point in Reachy's Cartesian space (torso frame).
131
132        The (x, y, z) coordinates are expressed in meters, where x is forward, y is left, and z is upward.
133
134        Args:
135            x: The x-coordinate of the target point.
136            y: The y-coordinate of the target point.
137            z: The z-coordinate of the target point.
138            duration: The time in seconds for the head to look at the point. Defaults to 2.0.
139            wait: Whether to wait for the movement to complete before returning. Defaults to False.
140            interpolation_mode: The interpolation mode for the movement, either "minimum_jerk" or "linear".
141                Defaults to "minimum_jerk".
142
143        Returns:
144            The unique GoToId associated with the movement command.
145
146        Raises:
147            ValueError: If the duration is set to 0.
148        """
149        if duration == 0:
150            raise ValueError("duration cannot be set to 0.")
151        if not self.neck.is_on():
152            self._logger.warning("head.neck is off. No command sent.")
153            return GoToId(id=-1)
154
155        request = GoToRequest(
156            cartesian_goal=CartesianGoal(
157                neck_cartesian_goal=NeckCartesianGoal(
158                    id=self._part_id,
159                    point=Point(x=x, y=y, z=z),
160                    duration=FloatValue(value=duration),
161                )
162            ),
163            interpolation_mode=get_grpc_interpolation_mode(interpolation_mode),
164        )
165        response = self._goto_stub.GoToCartesian(request)
166        if response.id == -1:
167            self._logger.error(f"Position {x}, {y}, {z} was not reachable. No command sent.")
168        elif wait:
169            self._wait_goto(response)
170        return response

180        """Send the neck to a specified roll, pitch, and yaw position.
181
182        The rotation is applied in the order: roll, pitch, yaw.
183
184        Args:
185            positions: A list of three float values representing the target roll, pitch, and yaw angles.
186            duration: The time in seconds for the neck to reach the target position. Defaults to 2.0.
187            wait: Whether to wait for the movement to complete before returning. Defaults to False.
188            interpolation_mode: The interpolation mode for the movement, either "minimum_jerk" or "linear".
189                Defaults to "minimum_jerk".
190            degrees: Whether the provided positions are in degrees. If True, positions will be converted to radians.
191                Defaults to True.
192
193        Returns:
194            The unique GoToId associated with the movement command.
195
196        Raises:
197            ValueError: If the duration is set to 0.
198        """
199        if duration == 0:
200            raise ValueError("duration cannot be set to 0.")
201        if not self.neck.is_on():
202            self._logger.warning("head.neck is off. No command sent.")

204
205        if degrees:
206            positions = np.deg2rad(positions).tolist()
207        request = GoToRequest(
208            joints_goal=JointsGoal(
209                neck_joint_goal=NeckJointGoal(
210                    id=self._part_id,
211                    joints_goal=NeckOrientation(
212                        rotation=Rotation3d(
213                            rpy=ExtEulerAngles(
214                                roll=FloatValue(value=positions[0]),
215                                pitch=FloatValue(value=positions[1]),
216                                yaw=FloatValue(value=positions[2]),
217                            )
218                        )
219                    ),
220                    duration=FloatValue(value=duration),
221                )
222            ),
223            interpolation_mode=get_grpc_interpolation_mode(interpolation_mode),
224        )

322    def send_goal_positions(self) -> None:
323        """Send goal positions to the head's joints.
324
325        If goal positions have been specified for any joint of the head, sends them to the robot.
326        """
327        if self.is_off():

329            return
330        for actuator in self._actuators.values():
331            actuator.send_goal_positions()

333    def goto_posture(
334        self,
335        duration: float = 2,
336        wait: bool = False,
337        wait_for_goto_end: bool = True,
338        interpolation_mode: str = "minimum_jerk",
339    ) -> GoToId:
340        """Send all neck joints to standard positions within the specified duration.
341
342        The default posture sets the neck joints to [0, -10, 0] (roll, pitch, yaw).
343
344        Args:
345            duration: The time in seconds for the neck to reach the target posture. Defaults to 2.
346            wait: Whether to wait for the movement to complete before returning. Defaults to False.
347            wait_for_goto_end: Whether to wait for all previous goto commands to finish before executing
348                the current command. If False, it cancels all ongoing commands. Defaults to True.
349            interpolation_mode: The interpolation mode for the movement, either "minimum_jerk" or "linear".
350                Defaults to "minimum_jerk".
351
352        Returns:
353            The unique GoToId associated with the movement command.
354
355        Raises:
356            ValueError: If the neck is off and the command cannot be sent.
357        """
358        if not wait_for_goto_end:
359            self.cancel_all_goto()
360        if self.neck.is_on():
361            return self.goto_joints([0, -10, 0], duration, wait, interpolation_mode)
362        else:
363            self._logger.warning("head.neck is off. No command sent.")
364        return GoToId(id=-1)
