reachy2_sdk.reachy_sdk

ReachySDK package.

This package provides remote access (via socket) to a Reachy robot. It automatically handles the synchronization with the robot. In particular, you can easily get an always up-to-date robot state (joint positions, sensors value). You can also send joint commands, compute forward or inverse kinematics.

 87    def connect(self) -> None:
 88        """Connects the SDK to the server."""
 89        if self._grpc_connected:
 90            self._logger.warning("Already connected to Reachy.")
 91            return
 92
 93        self._grpc_channel = grpc.insecure_channel(f"{self._host}:{self._sdk_port}")
 94
 95        self._stop_flag = threading.Event()
 96
 97        try:
 98            self._get_info()
 99        except ConnectionError:
100            self._logger.error(
101                f"Could not connect to Reachy with on IP address {self._host}, "
102                "check that the sdk server is running and that the IP is correct."
103            )
104            self._grpc_connected = False
105            return
106
107        self._setup_parts()
108        # self._setup_audio()
109        self._cameras = self._setup_video()
110
111        self._sync_thread = threading.Thread(target=self._start_sync_in_bg)
112        self._sync_thread.daemon = True
113        self._sync_thread.start()
114
115        self._audit_thread = threading.Thread(target=self._audit)
116        self._audit_thread.daemon = True
117        self._audit_thread.start()
118
119        self._grpc_connected = True
120        self._logger.info("Connected to Reachy.")

373    def _get_stream_update_loop(self, reachy_stub: reachy_pb2_grpc.ReachyServiceStub, freq: float) -> None:
374        """Update the state of the robot at a given frequency."""
375        stream_req = reachy_pb2.ReachyStreamStateRequest(id=self._robot.id, publish_frequency=freq)
376        try:
377            for state_update in reachy_stub.StreamReachyState(stream_req):
378                if self._l_arm is not None:
379                    self._l_arm._update_with(state_update.l_arm_state)
380                    if self._l_arm.gripper is not None:
381                        self._l_arm.gripper._update_with(state_update.l_hand_state)
382                if self._r_arm is not None:
383                    self._r_arm._update_with(state_update.r_arm_state)
384                    if self._r_arm.gripper is not None:
385                        self._r_arm.gripper._update_with(state_update.r_hand_state)
386                if self._head is not None:
387                    self._head._update_with(state_update.head_state)
388                if self._mobile_base is not None:
389                    self._mobile_base._update_with(state_update.mobile_base_state)
390        except grpc._channel._MultiThreadedRendezvous:
391            self._grpc_connected = False
392            raise ConnectionError(f"Connection with Reachy ip:{self._host} lost, check the sdk server status.")

544    def goto_posture(
545        self,
546        common_posture: str = "default",
547        wait: bool = False,
548        wait_for_goto_end: bool = True,
549        duration: float = 2,
550        interpolation_mode: str = "minimum_jerk",
551    ) -> GoToHomeId:
552        """Send all joints to standard positions in specified duration.
553
554        common_posture can be 'default' or 'elbow_90'.
555        Setting wait_for_goto_end to False will cancel all gotos on all parts and immediately send the commands.
556        Otherwise, the commands will be sent to a part when all gotos of its queue has been played.
557        """
558        if common_posture not in ["default", "elbow_90"]:
559            raise ValueError(f"common_posture {common_posture} not supported! Should be 'default' or 'elbow_90'")
560        head_id = None
561        r_arm_id = None
562        l_arm_id = None
563        if not wait_for_goto_end:
564            self.cancel_all_goto()
565        if self.head is not None:
566            is_last_commmand = self.r_arm is None and self.l_arm is None
567            wait_head = wait and is_last_commmand
568            head_id = self.head.goto_posture(
569                duration=duration, wait=wait_head, wait_for_goto_end=wait_for_goto_end, interpolation_mode=interpolation_mode
570            )
571        if self.r_arm is not None:
572            is_last_commmand = self.l_arm is None
573            wait_r_arm = wait and is_last_commmand
574            r_arm_id = self.r_arm.goto_posture(
575                common_posture,
576                duration=duration,
577                wait=wait_r_arm,
578                wait_for_goto_end=wait_for_goto_end,
579                interpolation_mode=interpolation_mode,
580            )
581        if self.l_arm is not None:
582            l_arm_id = self.l_arm.goto_posture(
583                common_posture,
584                duration=duration,
585                wait=wait,
586                wait_for_goto_end=wait_for_goto_end,
587                interpolation_mode=interpolation_mode,
588            )
589        ids = GoToHomeId(
590            head=head_id,
591            r_arm=r_arm_id,
592            l_arm=l_arm_id,
593        )
594        return ids
